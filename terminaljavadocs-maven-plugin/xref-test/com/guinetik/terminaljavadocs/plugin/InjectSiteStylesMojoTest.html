119      * When both exist, only staging should be processed.
120      *
121      * @throws Exception if reflection or execution fails
122      */
123     @Test
124     public void testPrefersStagingOverSite() throws Exception {
125         // Create both directories
126         File stagingDir = new File(tempFolder.getRoot(), "staging");
127         File siteDir = new File(tempFolder.getRoot(), "site");
128         stagingDir.mkdirs();
129         siteDir.mkdirs();
130 
131         createHtmlFile(new File(stagingDir, "staging-page.html"),
132                 "<html><head></head><body>staging</body></html>");
133         createHtmlFile(new File(siteDir, "site-page.html"),
134                 "<html><head></head><body>site</body></html>");
135 
136         mojo.execute();
137 
138         // Staging file should be processed
139         String stagingContent = readFile(new File(stagingDir, "staging-page.html"));
140         assertTrue("Staging file should be processed",
141                 stagingContent.contains("terminal-javadocs-injected"));
142 
143         // Site file should NOT be processed (staging takes priority)
144         String siteContent = readFile(new File(siteDir, "site-page.html"));
145         assertFalse("Site file should not be processed when staging exists",
146                 siteContent.contains("terminal-javadocs-injected"));
147     }

308     /**
309      * Verifies that already-injected files are skipped on re-execution.
310      * Prevents duplicate style injections.
311      *
312      * @throws Exception if reflection or execution fails
313      */
314     @Test
315     public void testSkipsAlreadyInjected() throws Exception {
316         File siteDir = new File(tempFolder.getRoot(), "site");
317         siteDir.mkdirs();
318 
319         // Create a file that already has the injection marker
320         String alreadyInjected = "<html><head><!-- terminal-javadocs-injected [site] -->" +
321                 "<link rel=\"stylesheet\" href=\"terminal-styles/terminaljavadocs-site.min.css\">" +
322                 "</head><body></body></html>";
323         createHtmlFile(new File(siteDir, "already-injected.html"), alreadyInjected);
324 
325         mojo.execute();
326 
327         String content = readFile(new File(siteDir, "already-injected.html"));
328         // Should have exactly one injection marker, not two
329         int firstIndex = content.indexOf("terminal-javadocs-injected");
330         int secondIndex = content.indexOf("terminal-javadocs-injected", firstIndex + 1);
331 
332         assertTrue("First marker should exist", firstIndex >= 0);
333         assertEquals("Should not have second marker (no re-injection)", -1, secondIndex);
334     }

444     /**
445      * Sets a private field on an object using reflection.
446      *
447      * @param target    the object to modify
448      * @param fieldName the name of the field to set
449      * @param value     the value to set
450      * @throws Exception if the field cannot be accessed or set
451      */
452     private void setField(Object target, String fieldName, Object value) throws Exception {
453         Field field = target.getClass().getDeclaredField(fieldName);
454         field.setAccessible(true);
455         field.set(target, value);
456     }

458     /**
459      * Creates an HTML file with the specified content.
460      *
461      * @param file    the file to create
462      * @param content the HTML content to write
463      * @throws IOException if the file cannot be written
464      */
465     private void createHtmlFile(File file, String content) throws IOException {
466         file.getParentFile().mkdirs();
467         Files.write(file.toPath(), content.getBytes(StandardCharsets.UTF_8));
468     }

470     /**
471      * Reads a file's content as a UTF-8 string.
472      *
473      * @param file the file to read
474      * @return the file content
475      * @throws IOException if the file cannot be read
476      */
477     private String readFile(File file) throws IOException {
478         return new String(Files.readAllBytes(file.toPath()), StandardCharsets.UTF_8);
